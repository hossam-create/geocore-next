## Listing Card Rules

Every card MUST have:
1. **Badge:** 🔨 AUCTION (red) or ⚡ BUY NOW (blue) — absolute top-left
2. **⭐ TOP** badge (yellow) if is_featured — absolute top-right
3. **Countdown timer** if auction (live, updates every second)
4. **Price** in primary blue color
5. **Location** with 📍 icon

```typescript
// ListingCard.tsx
export function ListingCard({ listing, onPress }) {
  const isAuction = listing.type === 'auction'

  return (
    <TouchableOpacity style={styles.card} onPress={onPress}>
      <View>
        <Image
          source={{ uri: listing.images?.[0]?.url || `https://picsum.photos/200/150?random=${listing.id}` }}
          style={styles.image}
        />
        {/* AUCTION or BUY NOW badge */}
        <View style={[styles.badge,
          { backgroundColor: isAuction ? '#E53935' : '#0071CE' }]}>
          <Text style={styles.badgeText}>
            {isAuction ? '🔨 AUCTION' : '⚡ BUY NOW'}
          </Text>
        </View>
        {listing.is_featured && (
          <View style={[styles.badge, styles.topRight, { backgroundColor: '#FFC220' }]}>
            <Text style={[styles.badgeText, { color: '#1A1A1A' }]}>⭐ TOP</Text>
          </View>
        )}
      </View>
      <View style={{ padding: 8 }}>
        <Text numberOfLines={2} style={styles.title}>{listing.title}</Text>
        <Text style={styles.price}>
          {listing.currency} {Number(listing.price).toLocaleString()}
        </Text>
        {isAuction && <CountdownTimer endsAt={listing.ends_at} />}
        <Text style={styles.location}>📍 {listing.city}</Text>
      </View>
    </TouchableOpacity>
  )
}
```

## Countdown Timer (Live)

```typescript
// components/CountdownTimer.tsx
export function CountdownTimer({ endsAt }) {
  const [timeLeft, setTimeLeft] = useState('')
  const [urgent, setUrgent] = useState(false)

  useEffect(() => {
    const tick = () => {
      const diff = new Date(endsAt).getTime() - Date.now()
      if (diff <= 0) { setTimeLeft('Ended'); return }
      const h = Math.floor(diff / 3600000)
      const m = Math.floor((diff % 3600000) / 60000)
      const s = Math.floor((diff % 60000) / 1000)
      setUrgent(diff < 300000)
      setTimeLeft(h > 0 ? `${h}h ${m}m` : `${m}m ${s}s`)
    }
    tick()
    const t = setInterval(tick, 1000)
    return () => clearInterval(t)
  }, [endsAt])

  return (
    <Text style={{ color: urgent ? '#E53935' : '#888', fontSize: 11 }}>
      ⏰ {timeLeft}
    </Text>
  )
}
```

## API Calls

```typescript
// src/api/client.ts
import axios from 'axios'
import * as SecureStore from 'expo-secure-store'

const api = axios.create({
  baseURL: 'https://geo-core-next.replit.app/api/v1',
  timeout: 10000,
})

api.interceptors.request.use(async (config) => {
  const token = await SecureStore.getItemAsync('access_token')
  if (token) config.headers.Authorization = `Bearer ${token}`
  return config
})

api.interceptors.response.use(
  (res) => res,
  async (error) => {
    if (error.response?.status === 401) {
      const refresh = await SecureStore.getItemAsync('refresh_token')
      if (refresh) {
        try {
          const { data } = await axios.post(
            'https://geo-core-next.replit.app/api/v1/auth/refresh',
            { refresh_token: refresh }
          )
          await SecureStore.setItemAsync('access_token', data.data.access_token)
          error.config.headers.Authorization = `Bearer ${data.data.access_token}`
          return api(error.config)
        } catch {
          await SecureStore.deleteItemAsync('access_token')
          await SecureStore.deleteItemAsync('refresh_token')
        }
      }
    }
    return Promise.reject(error)
  }
)

export default api
```
