# GeoCore Next — Complete Admin Dashboard
> ابعت لـ Replit Agent أو Cursor

---

## Mission

Build a **world-class admin dashboard** for GeoCore Next.

Inspiration: **Saleor Dashboard** (https://github.com/saleor/saleor-dashboard)
- Clean, modern React SPA
- Sidebar navigation with sections
- Data tables with filters + bulk actions
- Real-time stats with charts
- Full CRUD for everything

**Backend API:** `https://geo-core-next.replit.app/api/v1`
**Admin credentials:** role = "admin" in JWT

---

## Tech Stack

```
Framework:   React 18 + Vite
Language:    TypeScript strict
UI:          shadcn/ui + Tailwind CSS
Charts:      Recharts
Tables:      TanStack Table v8
State:       TanStack Query (server) + Zustand (local)
HTTP:        Axios with interceptors
Forms:       React Hook Form + Zod
Icons:       Lucide React
Routing:     React Router v6
```

```typescript
// src/pages/listings/ListingsPage.tsx
// Inspired by Saleor's product list page

export function ListingsPage() {
  const [status, setStatus] = useState("pending") // default: show pending first
  const [search, setSearch] = useState("")
  const [page, setPage] = useState(1)
  const [selected, setSelected] = useState<string[]>([])

  const { data, isLoading } = useQuery({
    queryKey: ["admin", "listings", status, search, page],
    queryFn: () => api.get(`/admin/listings?status=${status}&q=${search}&page=${page}&per_page=25`)
      .then(r => r.data),
  })

  // Bulk actions
  const bulkApprove = useMutation({
    mutationFn: (ids: string[]) => api.post("/admin/listings/bulk-approve", { ids }),
    onSuccess: () => { queryClient.invalidateQueries(["admin", "listings"]); setSelected([]) }
  })

  return (
    <div className="space-y-4">
      {/* Header */}
      <div className="flex items-center justify-between">
        <div>
          <h1 className="text-2xl font-bold">Listings</h1>
          <p className="text-gray-500 text-sm">{data?.meta?.total?.toLocaleString()} total listings</p>
        </div>
        <Button onClick={() => exportCSV()}>
          <Download className="w-4 h-4 mr-2" /> Export CSV
        </Button>
      </div>

      {/* Filters */}
      <Card className="p-4">
        <div className="flex flex-wrap gap-3">
          {/* Status tabs */}
          <div className="flex gap-1 bg-gray-100 p-1 rounded-lg">
            {["pending", "active", "sold", "expired", "rejected"].map(s => (
              <button
                key={s}
                onClick={() => setStatus(s)}
                className={`px-3 py-1.5 rounded-md text-sm font-medium capitalize transition-colors ${
                  status === s ? "bg-white shadow text-gray-900" : "text-gray-500 hover:text-gray-700"
                }`}
              >
                {s}
                {s === "pending" && data?.pending_count > 0 && (
                  <span className="ml-1.5 bg-red-500 text-white text-xs px-1.5 py-0.5 rounded-full">
                    {data.pending_count}
                  </span>
                )}
              </button>
            ))}
          </div>

          {/* Search */}
          <div className="flex-1 min-w-48">
            <Input
              placeholder="Search listings..."
              value={search}
              onChange={e => setSearch(e.target.value)}
              prefix={<Search className="w-4 h-4 text-gray-400" />}
            />
          </div>
        </div>
      </Card>

      {/* Bulk actions bar */}
      {selected.length > 0 && (
        <div className="bg-primary/10 border border-primary/20 rounded-lg p-3 flex items-center gap-3">
          <span className="text-primary text-sm font-medium">{selected.length} selected</span>
          <Button size="sm" onClick={() => bulkApprove.mutate(selected)}>
            <CheckCircle className="w-3 h-3 mr-1" /> Approve All
          </Button>
          <Button size="sm" variant="destructive" onClick={() => bulkReject(selected)}>
            <XCircle className="w-3 h-3 mr-1" /> Reject All
          </Button>
          <button onClick={() => setSelected([])} className="text-gray-500 text-sm ml-auto">
            Clear
          </button>
        </div>
      )}

      {/* Table */}
      <Card>
        <table className="w-full">
          <thead className="border-b">
            <tr>
              <th className="p-4 w-10">
                <input type="checkbox"
                  onChange={e => setSelected(e.target.checked ? data?.data?.map((l:any) => l.id) : [])}
                />
              </th>
              <th className="p-4 text-left text-sm font-medium text-gray-500">Listing</th>
              <th className="p-4 text-left text-sm font-medium text-gray-500">Seller</th>
              <th className="p-4 text-left text-sm font-medium text-gray-500">Category</th>
              <th className="p-4 text-left text-sm font-medium text-gray-500">Price</th>
              <th className="p-4 text-left text-sm font-medium text-gray-500">Type</th>
              <th className="p-4 text-left text-sm font-medium text-gray-500">Location</th>
              <th className="p-4 text-left text-sm font-medium text-gray-500">Date</th>
              <th className="p-4 text-left text-sm font-medium text-gray-500">Actions</th>
            </tr>
          </thead>
          <tbody>
            {data?.data?.map((listing: any) => (
              <tr key={listing.id} className="border-b hover:bg-gray-50 transition-colors">
                <td className="p-4">
                  <input type="checkbox"
                    checked={selected.includes(listing.id)}
                    onChange={e => {
                      if (e.target.checked) setSelected([...selected, listing.id])
                      else setSelected(selected.filter(id => id !== listing.id))
                    }}
                  />
                </td>
                <td className="p-4">
                  <div className="flex items-center gap-3">
                    <img
                      src={listing.images?.[0]?.url || `https://picsum.photos/40?random=${listing.id}`}
                      className="w-10 h-10 rounded-lg object-cover"
                    />
                    <div>
                      <p className="text-sm font-medium text-gray-900 line-clamp-1">{listing.title}</p>
                      <p className="text-xs text-gray-400">#{listing.id.slice(0, 8)}</p>
                    </div>
                  </div>
                </td>
                <td className="p-4 text-sm text-gray-600">{listing.user?.name}</td>
                <td className="p-4 text-sm text-gray-600">{listing.category?.name_en}</td>
                <td className="p-4 text-sm font-medium">
                  {listing.price ? `${listing.currency} ${Number(listing.price).toLocaleString()}` : "—"}
                </td>
                <td className="p-4">
                  <Badge variant={listing.type === "auction" ? "destructive" : "default"} className="text-xs">
                    {listing.type}
                  </Badge>
                </td>
                <td className="p-4 text-sm text-gray-500">{listing.city}, {listing.country}</td>
                <td className="p-4 text-xs text-gray-400">
                  {format(new Date(listing.created_at), "MMM d, yyyy")}
                </td>
                <td className="p-4">
                  <div className="flex gap-1">
                    <Button size="sm" variant="ghost" onClick={() => window.open(`/listings/${listing.id}`)}>
                      <Eye className="w-3 h-3" />
                    </Button>
                    {listing.status === "pending" && (
                      <>
                        <Button size="sm" className="bg-green-500 hover:bg-green-600 h-7 px-2"
                          onClick={() => approveListing(listing.id)}>
                          <Check className="w-3 h-3" />
                        </Button>
                        <Button size="sm" variant="destructive" className="h-7 px-2"
                          onClick={() => rejectListing(listing.id)}>
                          <X className="w-3 h-3" />
                        </Button>
                      </>
                    )}
                    <Button size="sm" variant="ghost" className="text-blue-500 h-7 px-2"
                      onClick={() => featureListing(listing.id)}>
                      <Star className="w-3 h-3" />
                    </Button>
                  </div>
                </td>
              </tr>
            ))}
          </tbody>
        </table>

        <div className="p-4 border-t">
          <Pagination meta={data?.meta} onChange={setPage} />
        </div>
      </Card>
    </div>
  )
}
```

### TASK-116: Final Value Fee Backend

```go
// In auctions/handler.go — when auction ends:
// If seller's price plan has final_value_fee_percent > 0:
// fee = current_bid * (fee_percent / 100)
// Deduct from seller's balance OR create an invoice

// Called by the auction end cron job
func (h *Handler) EndAuction(auction *Auction) error {
    // ...existing end logic...
    
    // Calculate final value fee
    if auction.CurrentBid > 0 {
        var plan pricing.PricePlan
        h.db.Joins("JOIN users ON users.user_group_id = price_plans.user_group_id").
            Where("users.id = ?", auction.SellerID).
            First(&plan)
        
        if plan.FinalValueFeePercent > 0 {
            fee := auction.CurrentBid * (plan.FinalValueFeePercent / 100)
            wallet.Debit(h.db, auction.SellerID, fee, "final_value_fee",
                auction.ID.String(), 
                fmt.Sprintf("Final value fee (%.1f%%)", plan.FinalValueFeePercent),
                auction.Currency)
        }
    }
}
```

```go
// GET /feeds/listings.rss
// GET /feeds/auctions.rss

func (h *Handler) ListingsRSS(c *gin.Context) {
    var listings []Listing
    h.db.Where("status = ?", "active").
        Order("created_at DESC").
        Limit(50).
        Find(&listings)

    c.Header("Content-Type", "application/rss+xml")
    c.Writer.WriteString(`<?xml version="1.0" encoding="UTF-8"?>`)
    c.Writer.WriteString(`<rss version="2.0">`)
    c.Writer.WriteString(`<channel>`)
    c.Writer.WriteString(fmt.Sprintf("<title>GeoCore Listings</title>"))
    c.Writer.WriteString(fmt.Sprintf("<link>%s</link>", os.Getenv("FRONTEND_URL")))

    for _, l := range listings {
        price := ""
        if l.Price != nil { price = fmt.Sprintf("%.2f %s", *l.Price, l.Currency) }
        c.Writer.WriteString(fmt.Sprintf(`
            <item>
                <title><![CDATA[%s]]></title>
                <link>%s/listings/%s</link>
                <description><![CDATA[%s — %s — %s, %s]]></description>
                <pubDate>%s</pubDate>
                <guid>%s</guid>
            </item>`,
            l.Title,
            os.Getenv("FRONTEND_URL"), l.ID,
            l.Description, price, l.City, l.Country,
            l.CreatedAt.Format(time.RFC1123Z),
            l.ID,
        ))
    }
    c.Writer.WriteString(`</channel></rss>`)
}
```
