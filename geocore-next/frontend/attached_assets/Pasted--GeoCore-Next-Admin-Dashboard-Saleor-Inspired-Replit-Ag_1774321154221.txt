> ابعت لـ Replit Agent

---

## Mission

Build a **complete admin dashboard** for GeoCore Next.

Use **Saleor Dashboard** as your visual and UX reference:
- Read the source code at: https://github.com/saleor/saleor-dashboard
- Study how they structure pages, tables, forms, and navigation
- Copy the **design patterns and UX** — NOT the GraphQL/Saleor-specific code
- Connect everything to **our REST API** at: `https://geo-core-next.replit.app/api/v1`

Read these files from Saleor to understand their patterns:

```
https://raw.githubusercontent.com/saleor/saleor-dashboard/main/src/components/AppLayout/AppLayout.tsx
https://raw.githubusercontent.com/saleor/saleor-dashboard/main/src/components/TableCellHeader/TableCellHeader.tsx

https://raw.githubusercontent.com/saleor/saleor-dashboard/main/src/orders/components/OrderList/OrderList.tsx
```

From Saleor, adopt these UX patterns:
- **Left sidebar** with dark background + sections + icons
- **Data tables** with checkboxes, sortable columns, bulk actions
- **Filter chips** above tables (status, date range, category)
- **Detail pages** with card sections (split left/right layout)
- **Breadcrumb navigation**
- **Save bar** at bottom of forms (sticky with Cancel/Save)
- **Status badges** with colors (active=green, pending=yellow, rejected=red)
- **Confirmation dialogs** for destructive actions
- **Empty states** with illustrations and CTA

Saleor shows: activity summary + quick stats + recent orders.
Adapt to: activity summary + quick stats + pending approvals.

```tsx
export function DashboardPage() {
  // Auto-refresh stats every 30 seconds
```

## Step 7 — Listings List Page (Saleor Product List Pattern)

```tsx
// Saleor's product list has:
// - Filter bar with status chips
// - Searchable, sortable table
// - Checkbox + bulk actions
// - Row click → detail page

export function ListingsListPage() {
  const [filters, setFilters] = useState({
    status: "pending",
    search: "",
    category: "",
    type: "",
    page: 1,
  })
  const [selectedIds, setSelectedIds] = useState<string[]>([])

  const { data, isLoading } = useQuery({
    queryKey: ["admin", "listings", filters],
    queryFn: () => adminApi.getListings(filters),
    keepPreviousData: true,
  })

  return (
    <PageLayout
      title="Listings"
      subtitle={`${data?.meta?.total?.toLocaleString() ?? "—"} total`}
      actions={
        <Button variant="outline" onClick={exportCSV}>
          <Download className="w-4 h-4 mr-2" />
          Export CSV
        </Button>
      }
    >
      {/* Filter chips — Saleor style */}
      <Card className="mb-4">
        <CardContent className="p-4">
          <div className="flex flex-wrap gap-3 items-center">

            {/* Status tabs */}
            <div className="flex rounded-lg border divide-x overflow-hidden">
              {["pending","active","sold","expired","rejected"].map(s => (
                <button
                  key={s}
                  onClick={() => setFilters(f => ({ ...f, status: s, page: 1 }))}
                  className={`px-4 py-2 text-sm font-medium capitalize transition-colors ${
                    filters.status === s
                      ? "bg-primary text-white"
                      : "hover:bg-gray-50 text-gray-600"
                  }`}
                >
                  {s}
                  {s === "pending" && data?.pending_count > 0 && (
                    <span className="ml-1.5 bg-red-500 text-white text-xs
                      rounded-full px-1.5 py-0.5 font-bold">
                      {data.pending_count}
                    </span>
                  )}
                </button>
              ))}
            </div>

            {/* Search */}
            <div className="relative flex-1 min-w-56">
              <Search className="absolute left-3 top-2.5 w-4 h-4 text-gray-400" />
              <Input
                className="pl-9"
                placeholder="Search listings..."
                value={filters.search}
                onChange={e => setFilters(f => ({ ...f, search: e.target.value }))}
              />
            </div>

            {/* Type filter */}
            <Select value={filters.type}
              onValueChange={v => setFilters(f => ({ ...f, type: v }))}>
              <SelectTrigger className="w-36">
                <SelectValue placeholder="All types" />
              </SelectTrigger>
              <SelectContent>
                <SelectItem value="">All types</SelectItem>
                <SelectItem value="sell">Sell</SelectItem>
                <SelectItem value="auction">Auction</SelectItem>
                <SelectItem value="dutch">Dutch</SelectItem>
                <SelectItem value="reverse">Reverse</SelectItem>
              </SelectContent>
            </Select>

          </div>
        </CardContent>
      </Card>

      {/* Bulk actions bar — appears when items selected */}
      {selectedIds.length > 0 && (
        <BulkActionsBar
          count={selectedIds.length}
          onClear={() => setSelectedIds([])}
          actions={[
            {
              label: "Approve",
              icon: CheckCircle,
              variant: "success",
              onClick: () => bulkApprove(selectedIds),
            },
            {
              label: "Reject",
              icon: XCircle,
              variant: "destructive",
              onClick: () => setRejectDialogOpen(true),
            },
            {
              label: "Feature",
              icon: Star,
              onClick: () => bulkFeature(selectedIds),
            },
          ]}
        />
      )}

      {/* Main table */}
      <Card>
        <Table>
          <TableHeader>
            <TableRow>
              <TableHead className="w-10">
                <Checkbox
                  checked={selectedIds.length === data?.data?.length}
                  onCheckedChange={checked =>
                    setSelectedIds(checked
                      ? data?.data?.map((l:any) => l.id) ?? []
                      : [])
                  }
                />
              </TableHead>
              <TableHead>Listing</TableHead>
              <TableHead>Seller</TableHead>
              <TableHead>Category</TableHead>
              <TableHead>Price</TableHead>
              <TableHead>Type</TableHead>
              <TableHead>Location</TableHead>
              <TableHead>Date</TableHead>
              <TableHead className="w-32">Actions</TableHead>
            </TableRow>
          </TableHeader>

          <TableBody>
            {isLoading ? (
              <TableRow>
                <TableCell colSpan={9}>
                  <LoadingSkeleton rows={10} />
                </TableCell>
              </TableRow>
            ) : data?.data?.length === 0 ? (
              <TableRow>
                <TableCell colSpan={9}>
                  <EmptyState
                    icon={Tag}
                    title="No listings found"
                    description="Try changing the filters"
                  />
                </TableCell>
              </TableRow>
            ) : data?.data?.map((listing: any) => (
              <TableRow
                key={listing.id}
                className="cursor-pointer hover:bg-gray-50"
                onClick={() => navigate(`/admin/listings/${listing.id}`)}
              >
                <TableCell onClick={e => e.stopPropagation()}>
                  <Checkbox
                    checked={selectedIds.includes(listing.id)}
                    onCheckedChange={checked => {
                      setSelectedIds(prev =>
                        checked
                          ? [...prev, listing.id]
                          : prev.filter(id => id !== listing.id)
                      )
                    }}
                  />
                </TableCell>

                <TableCell>
                  <div className="flex items-center gap-3">
                    <img
                      src={listing.images?.[0]?.url
                        || `https://picsum.photos/48?random=${listing.id}`}
                      className="w-12 h-12 rounded-lg object-cover shrink-0"
                    />
                    <div>
                      <p className="font-medium text-sm line-clamp-1 max-w-48">
                        {listing.title}
                      </p>
                      <p className="text-xs text-gray-400 font-mono">
                        #{listing.id.slice(0,8)}
                      </p>
                    </div>
                  </div>
                </TableCell>

                <TableCell className="text-sm">{listing.user?.name}</TableCell>
                <TableCell className="text-sm">{listing.category?.name_en}</TableCell>

                <TableCell className="text-sm font-semibold">
                  {listing.price
                    ? `${listing.currency} ${Number(listing.price).toLocaleString()}`
                    : <span className="text-gray-400">—</span>
                  }
                </TableCell>

                <TableCell>
                  <TypeBadge type={listing.type} />
                </TableCell>

                <TableCell className="text-sm text-gray-500">
                  {listing.city}, {listing.country}
                </TableCell>

                <TableCell className="text-xs text-gray-400">
                  {format(new Date(listing.created_at), "MMM d, yyyy")}
                </TableCell>

                <TableCell onClick={e => e.stopPropagation()}>
                  <div className="flex gap-1">
                    {listing.status === "pending" && (
                      <>
                        <Button size="icon" variant="ghost"
                          className="h-8 w-8 text-green-600 hover:bg-green-50"
                          onClick={() => approveListing(listing.id)}>
                          <Check className="w-4 h-4" />
                        </Button>
                        <Button size="icon" variant="ghost"
                          className="h-8 w-8 text-red-600 hover:bg-red-50"
                          onClick={() => openRejectDialog(listing.id)}>
                          <X className="w-4 h-4" />
                        </Button>
                      </>
                    )}
                    <Button size="icon" variant="ghost"
                      className="h-8 w-8 text-yellow-600 hover:bg-yellow-50"
                      onClick={() => toggleFeatured(listing)}>
                      <Star className={`w-4 h-4 ${listing.is_featured ? "fill-yellow-400" : ""}`} />
                    </Button>
                    <DropdownMenu>
                      <DropdownMenuTrigger asChild>
                        <Button size="icon" variant="ghost" className="h-8 w-8">
                          <MoreHorizontal className="w-4 h-4" />
                        </Button>
                      </DropdownMenuTrigger>
                      <DropdownMenuContent align="end">
                        <DropdownMenuItem onClick={() => window.open(`/listings/${listing.slug || listing.id}`)}>
                          <Eye className="w-4 h-4 mr-2" /> View on site
                        </DropdownMenuItem>
                        <DropdownMenuItem className="text-red-600"
                          onClick={() => deleteListing(listing.id)}>
                          <Trash2 className="w-4 h-4 mr-2" /> Delete
                        </DropdownMenuItem>
                      </DropdownMenuContent>
                    </DropdownMenu>
                  </div>
                </TableCell>
              </TableRow>
            ))}
          </TableBody>
        </Table>

        <div className="p-4 border-t">
          <Pagination
            page={filters.page}
            total={data?.meta?.total ?? 0}
            perPage={25}
            onChange={page => setFilters(f => ({ ...f, page }))}
          />
        </div>
      </Card>
    </PageLayout>
  )
}
```

## Step 8 — Listing Detail Page (Saleor Product Detail Pattern)

Saleor shows product details in a 2-column layout:
- Left (wide): main info, description, images
- Right (narrow): status, category, metadata

```tsx
export function ListingDetailPage() {
  const { id } = useParams()
  const { data: listing } = useQuery({
    queryKey: ["admin", "listing", id],
    queryFn: () => adminApi.getListing(id!),
  })

  return (
    <PageLayout
      title={listing?.title ?? "Listing"}
      breadcrumbs={[
        { label: "Listings", href: "/admin/listings" },
        { label: listing?.title ?? "..." }
      ]}
    >
      <div className="grid grid-cols-1 xl:grid-cols-3 gap-6">

        {/* Left column — main content */}
        <div className="xl:col-span-2 space-y-6">

          {/* General info */}
          <Card>
            <CardHeader><CardTitle>Listing Information</CardTitle></CardHeader>
            <CardContent className="space-y-4">
              <div>
                <Label>Title</Label>
                <p className="text-gray-900 mt-1">{listing?.title}</p>
              </div>
              <div>
                <Label>Description</Label>
                <p className="text-gray-600 mt-1 text-sm leading-relaxed">
                  {listing?.description}
                </p>
              </div>
              <div className="grid grid-cols-2 gap-4">
                <div>
                  <Label>Price</Label>
                  <p className="font-bold text-lg text-primary mt-1">
                    {listing?.currency} {Number(listing?.price).toLocaleString()}
                  </p>
                </div>
                <div>
                  <Label>Condition</Label>
                  <p className="mt-1">{listing?.condition}</p>
                </div>
              </div>
            </CardContent>
          </Card>

          {/* Images */}
          <Card>
            <CardHeader><CardTitle>Images ({listing?.images?.length ?? 0})</CardTitle></CardHeader>
            <CardContent>
              <div className="grid grid-cols-4 gap-3">
                {listing?.images?.map((img: any) => (
                  <div key={img.id} className="relative aspect-square rounded-lg overflow-hidden">
                    <img src={img.url} className="w-full h-full object-cover" />
                    {img.is_cover && (
                      <span className="absolute top-1 left-1 bg-primary text-white
                        text-xs px-1.5 py-0.5 rounded">Cover</span>
                    )}
                  </div>
                ))}
              </div>
            </CardContent>
          </Card>

          {/* Custom fields */}
          {listing?.attributes && Object.keys(listing.attributes).length > 0 && (
            <Card>
              <CardHeader><CardTitle>Details</CardTitle></CardHeader>
              <CardContent>
                <dl className="grid grid-cols-2 gap-3">
                  {Object.entries(listing.attributes).map(([key, val]) => (
                    <div key={key}>
                      <dt className="text-xs text-gray-500 capitalize">
                        {key.replace(/_/g, " ")}
                      </dt>
                      <dd className="text-sm font-medium mt-0.5">{String(val)}</dd>
                    </div>
                  ))}
                </dl>
              </CardContent>
            </Card>
          )}

        </div>

        {/* Right column — metadata */}
        <div className="space-y-6">

          {/* Status + Actions */}
          <Card>
            <CardHeader><CardTitle>Status</CardTitle></CardHeader>
            <CardContent className="space-y-3">
              <StatusBadge status={listing?.status} size="lg" />

              {listing?.status === "pending" && (
                <div className="flex gap-2 pt-2">
                  <Button className="flex-1 bg-green-600 hover:bg-green-700"
                    onClick={() => approve()}>
                    <Check className="w-4 h-4 mr-1" /> Approve

                  <Button variant="destructive" className="flex-1"

          </Card>

          {/* Seller */}
          <Card>

              <div className="flex justify-between">

                <span className="font-medium">{listing?.city}</span>
              </div>
              {listing?.address && (
                <div className="flex justify-between">
                  <span className="text-gray-500">Address</span>
                  <span className="font-medium text-right">{listing?.address}</span>
                </div>
              )}
            </CardContent>
          </Card>

          {/* Timeline */}
          <Card>
            <CardHeader><CardTitle>Timeline</CardTitle></CardHeader>
            <CardContent className="space-y-2 text-sm">
              <div className="flex justify-between">
                <span className="text-gray-500">Created</span>
                <span>{format(new Date(listing?.created_at), "MMM d, yyyy")}</span>
              </div>
              <div className="flex justify-between">
                <span className="text-gray-500">Updated</span>
                <span>{format(new Date(listing?.updated_at), "MMM d, yyyy")}</span>
              </div>
              {listing?.expires_at && (
                <div className="flex justify-between">
                  <span className="text-gray-500">Expires</span>
                  <span>{format(new Date(listing?.expires_at), "MMM d, yyyy")}</span>
                </div>
              )}
              <div className="flex justify-between">
                <span className="text-gray-500">Views</span>
                <span>{listing?.view_count?.toLocaleString()}</span>
              </div>
            </CardContent>
          </Card>

        </div>
      </div>
    </PageLayout>
  )
}
```

```typescript
// src/api/client.ts
import axios from "axios"

const api = axios.create({
  baseURL: "https://geo-core-next.replit.app/api/v1",
})

api.interceptors.request.use(config => {
  const token = localStorage.getItem("admin_token")
  if (token) config.headers.Authorization = `Bearer ${token}`
  return config
})

api.interceptors.response.use(
  res => res,
  async err => {
    if (err.response?.status === 401) {
      localStorage.removeItem("admin_token")
      window.location.href = "/admin/login"
    }
    return Promise.reject(err)
  }
)

export default api

// Admin-specific API calls
export const adminApi = {
  getStats: () => api.get("/admin/stats").then(r => r.data.data),
  getListings: (params: any) => api.get("/admin/listings", { params }).then(r => r.data),
  getListing: (id: string) => api.get(`/admin/listings/${id}`).then(r => r.data.data),
  approveListing: (id: string) => api.post(`/admin/listings/${id}/approve`),
  rejectListing: (id: string, reason: string) =>
    api.post(`/admin/listings/${id}/reject`, { reason }),
  featureListing: (id: string) => api.post(`/admin/listings/${id}/feature`),
  getUsers: (params: any) => api.get("/admin/users", { params }).then(r => r.data),
  blockUser: (id: string) => api.post(`/admin/users/${id}/block`),
  unblockUser: (id: string) => api.post(`/admin/users/${id}/unblock`),
  getReports: (params: any) => api.get("/admin/reports", { params }).then(r => r.data),
  resolveReport: (id: string) => api.post(`/admin/reports/${id}/resolve`),
```
