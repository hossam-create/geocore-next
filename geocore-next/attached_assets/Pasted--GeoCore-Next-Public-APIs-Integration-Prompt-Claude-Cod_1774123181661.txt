### API 1 — Currency Exchange (Frankfurter)
**URL:** `https://api.frankfurter.app`
**No API key needed**
**Docs:** https://www.frankfurter.app/docs

**Where to use:**
- New file: `backend/pkg/currency/currency.go`
- Called from: `backend/internal/listings/handler.go` when creating/updating a listing

**What to build:**

```go
// backend/pkg/currency/currency.go
package currency

import (
    "encoding/json"
    "fmt"
    "net/http"
    "time"
)

var httpClient = &http.Client{Timeout: 5 * time.Second}

// SupportedCurrencies returns list of all supported currency codes
func SupportedCurrencies() ([]string, error) {
    resp, err := httpClient.Get("https://api.frankfurter.app/currencies")
    if err != nil {
        return nil, fmt.Errorf("currency.SupportedCurrencies: %w", err)
    }
    defer resp.Body.Close()
    var result map[string]string
    if err := json.NewDecoder(resp.Body).Decode(&result); err != nil {
        return nil, fmt.Errorf("currency.SupportedCurrencies decode: %w", err)
    }
    codes := make([]string, 0, len(result))
    for code := range result {
        codes = append(codes, code)
    }
    return codes, nil
}

// Convert converts an amount from one currency to another
// Returns converted amount
func Convert(amount float64, from, to string) (float64, error) {
    if from == to {
        return amount, nil
    }
    url := fmt.Sprintf("https://api.frankfurter.app/latest?amount=%.2f&from=%s&to=%s", amount, from, to)
    resp, err := httpClient.Get(url)
    if err != nil {
        return 0, fmt.Errorf("currency.Convert: %w", err)
    }
    defer resp.Body.Close()
    var result struct {
        Rates map[string]float64 `json:"rates"`
    }
    if err := json.NewDecoder(resp.Body).Decode(&result); err != nil {
        return 0, fmt.Errorf("currency.Convert decode: %w", err)
    }
    converted, ok := result.Rates[to]
    if !ok {
        return 0, fmt.Errorf("currency.Convert: currency %s not found", to)
    }
    return converted, nil
}
```

**Add new endpoint:**
```
GET /api/v1/currencies
```
Returns list of supported currencies — used by frontend dropdowns.

```go
// In listings/handler.go or a new currencies handler
func (h *Handler) GetCurrencies(c *gin.Context) {
    codes, err := currency.SupportedCurrencies()
    if err != nil {
        response.InternalError(c, err)
        return
    }
    response.OK(c, codes)
}
```

### API 2 — IP Geolocation (ip-api.com)
**URL:** `http://ip-api.com/json/{ip}`
**No API key needed** (free for non-commercial, 45 req/min)
**Docs:** https://ip-api.com/docs

**Where to use:**
- New file: `backend/pkg/geo/ipgeo.go`
- Called from: `backend/internal/users/handler.go` on first login/register
- Auto-fill user's country and city

**What to build:**

```go
// backend/pkg/geo/ipgeo.go
package geo

import (
    "encoding/json"
    "fmt"
    "net/http"
    "time"
)

var httpClient = &http.Client{Timeout: 5 * time.Second}

type IPInfo struct {
    Country     string  `json:"country"`
    CountryCode string  `json:"countryCode"`
    City        string  `json:"city"`
    Lat         float64 `json:"lat"`
    Lon         float64 `json:"lon"`
    Currency    string  `json:"currency"` // not in ip-api, but we can infer
    Timezone    string  `json:"timezone"`
}

func LookupIP(ip string) (*IPInfo, error) {
    // Skip for localhost
    if ip == "127.0.0.1" || ip == "::1" {
        return &IPInfo{Country: "Unknown", CountryCode: "US", City: "Unknown"}, nil
    }
    url := fmt.Sprintf("http://ip-api.com/json/%s?fields=country,countryCode,city,lat,lon,timezone", ip)
    resp, err := httpClient.Get(url)
    if err != nil {
        return nil, fmt.Errorf("geo.LookupIP: %w", err)
    }
    defer resp.Body.Close()
    var info IPInfo
    if err := json.NewDecoder(resp.Body).Decode(&info); err != nil {
        return nil, fmt.Errorf("geo.LookupIP decode: %w", err)
    }
    return &info, nil
}
```

**Add new endpoint:**
```
GET /api/v1/detect-location
```
Frontend calls this on page load to pre-fill country/city in forms.

```go
func (h *Handler) DetectLocation(c *gin.Context) {
    ip := c.ClientIP()
    info, err := geo.LookupIP(ip)
    if err != nil {
        response.OK(c, gin.H{"country": "", "city": ""})
        return
    }
    response.OK(c, info)
}
```

### API 3 — Geocoding (Nominatim / OpenStreetMap)
**URL:** `https://nominatim.openstreetmap.org`
**No API key needed** (add User-Agent header as required)
**Docs:** https://nominatim.org/release-docs/develop/api/Search/

**Where to use:**
- Add to `backend/pkg/geo/geocode.go`
- Called from: `backend/internal/listings/handler.go` when creating a listing with address but no lat/lng

**What to build:**

```go
// backend/pkg/geo/geocode.go
package geo

import (
    "encoding/json"
    "fmt"
    "net/http"
    "net/url"
    "time"
)

type GeoResult struct {
    Lat string `json:"lat"`
    Lon string `json:"lon"`
}

func GeocodeAddress(address, city, country string) (*GeoResult, error) {
    query := fmt.Sprintf("%s, %s, %s", address, city, country)
    apiURL := fmt.Sprintf(
        "https://nominatim.openstreetmap.org/search?q=%s&format=json&limit=1",
        url.QueryEscape(query),
    )
    req, _ := http.NewRequest("GET", apiURL, nil)
    // Nominatim requires a User-Agent
    req.Header.Set("User-Agent", "GeoCore-Next/1.0 (geocore@example.com)")

    resp, err := httpClient.Do(req)
    if err != nil {
        return nil, fmt.Errorf("geo.GeocodeAddress: %w", err)
    }
    defer resp.Body.Close()
    var results []GeoResult
    if err := json.NewDecoder(resp.Body).Decode(&results); err != nil {
        return nil, fmt.Errorf("geo.GeocodeAddress decode: %w", err)
    }
    if len(results) == 0 {
        return nil, fmt.Errorf("geo.GeocodeAddress: no results for %q", query)
    }
    return &results[0], nil
}
```

**Update listings/handler.go Create():**
```go

    if result, err := geo.GeocodeAddress(req.Address, req.City, req.Country); err == nil {
```

### API 4 — Email Validation (EVA)
**URL:** `https://eva.pingutil.com/email?email={email}`
**No API key needed**
**Docs:** https://eva.pingutil.com

**Where to use:**
- New file: `backend/pkg/validation/email.go`
- Called from: `backend/internal/auth/handler.go` in Register()

**What to build:**

```go
// backend/pkg/validation/email.go
package validation

import (
    "encoding/json"
    "fmt"
    "net/http"
    "time"
)

var httpClient = &http.Client{Timeout: 4 * time.Second}

type EmailResult struct {
    Valid       bool   `json:"valid"`
    Disposable bool   `json:"disposable"`
    Domain      string `json:"domain"`
}

// ValidateEmail checks if email is real and not disposable
// Returns (isValid, isDisposable, error)
func ValidateEmail(email string) (bool, bool, error) {
    url := fmt.Sprintf("https://eva.pingutil.com/email?email=%s", email)
    resp, err := httpClient.Get(url)
    if err != nil {
        // Don't block registration if API is down
        return true, false, nil
    }
    defer resp.Body.Close()
    var result struct {
        Data EmailResult `json:"data"`
    }
    if err := json.NewDecoder(resp.Body).Decode(&result); err != nil {
        return true, false, nil
    }
    return result.Data.Valid, result.Data.Disposable, nil
}
```

**Update auth/handler.go Register():**
```go

if valid, disposable, _ := validation.ValidateEmail(req.Email); !valid {
    response.BadRequest(c, "Invalid email address")
    return
}
if disposable {
    response.BadRequest(c, "Temporary email addresses are not allowed")
    return
}
```

**URL:** `http://apilayer.net/api/validate?access_key=FREE_KEY&number={phone}`
**Free tier:** 250 requests/month
**Alternative (no key):** Use regex validation as fallback

**Where to use:**
- Add to `backend/pkg/validation/phone.go`
- Called from: `backend/internal/users/handler.go` UpdateMe()

**What to build:**

```go
// backend/pkg/validation/phone.go
package validation

import "regexp"

// ValidatePhoneFormat does basic E.164 format check
// E.164: +[country code][number], e.g. +201234567890
func ValidatePhoneFormat(phone string) bool {
    if phone == "" {
        return true // optional field
    }
    re := regexp.MustCompile(`^\+[1-9]\d{7,14}$`)
    return re.MatchString(phone)
}
```

**Update users/handler.go UpdateMe():**
```go
if req.Phone != "" && !validation.ValidatePhoneFormat(req.Phone) {

    return
}
```

### API 6 — Public Holidays (Nager.Date)
**URL:** `https://date.nager.at/api/v3/PublicHolidays/{year}/{countryCode}`
**No API key needed**
**Docs:** https://date.nager.at/swagger/index.html

**Where to use:**
- New file: `backend/pkg/calendar/holidays.go`
- Used by: Auction scheduler — don't end auctions on public holidays (optional feature)
- Also useful for: showing "X listings expire during holiday season" warning

**What to build:**

```go
// backend/pkg/calendar/holidays.go
package calendar

import (
    "encoding/json"
    "fmt"
    "net/http"
    "time"
)

var httpClient = &http.Client{Timeout: 5 * time.Second}

type Holiday struct {
    Date        string `json:"date"`
    LocalName   string `json:"localName"`
    Name        string `json:"name"`
    CountryCode string `json:"countryCode"`
}

func GetHolidays(year int, countryCode string) ([]Holiday, error) {
    url := fmt.Sprintf("https://date.nager.at/api/v3/PublicHolidays/%d/%s", year, countryCode)
    resp, err := httpClient.Get(url)
    if err != nil {
        return nil, fmt.Errorf("calendar.GetHolidays: %w", err)
    }
    defer resp.Body.Close()
    var holidays []Holiday
    if err := json.NewDecoder(resp.Body).Decode(&holidays); err != nil {
        return nil, fmt.Errorf("calendar.GetHolidays decode: %w", err)
    }
    return holidays, nil
}

func IsHoliday(date time.Time, countryCode string) (bool, error) {
    holidays, err := GetHolidays(date.Year(), countryCode)
    if err != nil {
        return false, err
    }
    dateStr := date.Format("2006-01-02")
    for _, h := range holidays {
        if h.Date == dateStr {
            return true, nil
        }
    }
    return false, nil
}
```

## Important Rules

- **Never block a request** if a public API is down — always have a fallback:
  ```go
  info, err := geo.LookupIP(ip)
  if err != nil {
      // Log it, but don't fail the request
      logger.Warn("IP geo lookup failed", zap.Error(err))
      info = &geo.IPInfo{} // empty defaults
  }
  ```

- **Cache API responses** where possible using Redis:
  ```go
  // Cache currency list for 1 hour
  cacheKey := "currencies:list"
  // Try Redis first, then hit API, then cache result
  ```

- **Respect rate limits:**
  - ip-api.com: 45 req/min → add Redis rate limit tracking
  - Nominatim: 1 req/sec → add `time.Sleep(1 * time.Second)` in batch operations
  - Frankfurter: no strict limit but cache results

- **Timeouts:** All HTTP clients must have a timeout (already included in examples above — 4-5 seconds max)
