```typescript
// src/screens/LoginScreen.tsx
export function LoginScreen({ navigation }) {
  const { login } = useAuthStore()
  const [email, setEmail] = useState('')
  const [password, setPassword] = useState('')
  const [loading, setLoading] = useState(false)
  const [error, setError] = useState('')

  const handleLogin = async () => {
    if (!email || !password) {
      setError('Please fill in all fields')
      return
    }
    setLoading(true)
    setError('')
    try {
      await login(email, password)
      // Navigation handled by root navigator
    } catch (e: any) {
      setError(e.response?.data?.error || 'Login failed')
    } finally {
      setLoading(false)
    }
  }

  return (
    <SafeAreaView style={{ flex: 1, backgroundColor: Colors.white }}>
      <KeyboardAvoidingView style={{ flex: 1 }} behavior="padding">

        {/* Header */}
        <View style={{
          backgroundColor: Colors.primary,
          paddingVertical: 40,
          alignItems: 'center',
        }}>
          <Text style={{ fontSize: 32, fontWeight: 'bold', color: Colors.secondary }}>
            GeoCore
          </Text>
          <Text style={{ color: 'rgba(255,255,255,0.8)', marginTop: 4 }}>
            Buy. Sell. Bid. Win.
          </Text>
        </View>

        <ScrollView style={{ padding: 24 }}>
          <Text style={{ fontSize: 24, fontWeight: 'bold', marginBottom: 24, color: Colors.dark }}>
            Sign In
          </Text>

          {error ? (
            <View style={{
              backgroundColor: '#FFEBEE', borderRadius: 8,
              padding: 12, marginBottom: 16,
            }}>
              <Text style={{ color: Colors.danger }}>{error}</Text>
            </View>
          ) : null}

          {/* Email */}
          <Text style={styles.label}>Email</Text>
          <TextInput
            style={styles.input}
            value={email}
            onChangeText={setEmail}
            placeholder="your@email.com"
            keyboardType="email-address"
            autoCapitalize="none"
          />

          {/* Password */}
          <Text style={styles.label}>Password</Text>
          <TextInput
            style={styles.input}
            value={password}
            onChangeText={setPassword}
            placeholder="••••••••"
            secureTextEntry
          />

          {/* Login Button */}
          <TouchableOpacity
            style={{
              backgroundColor: Colors.primary,
              padding: 16, borderRadius: 8,
              alignItems: 'center', marginTop: 8,
              opacity: loading ? 0.7 : 1,
            }}
            onPress={handleLogin}
            disabled={loading}
          >
            {loading
              ? <ActivityIndicator color={Colors.white} />
              : <Text style={{ color: Colors.white, fontSize: 16, fontWeight: 'bold' }}>
                  Sign In
                </Text>
            }
          </TouchableOpacity>

          {/* Register Link */}
          <TouchableOpacity
            style={{ alignItems: 'center', marginTop: 24 }}
            onPress={() => navigation.navigate('Register')}
          >
            <Text style={{ color: Colors.gray }}>
              Don't have an account?{' '}
              <Text style={{ color: Colors.primary, fontWeight: 'bold' }}>Sign Up</Text>
            </Text>
          </TouchableOpacity>

          {/* Continue as Guest */}
          <TouchableOpacity
            style={{ alignItems: 'center', marginTop: 12 }}
            onPress={() => navigation.navigate('Main')}
          >
            <Text style={{ color: Colors.gray }}>Continue as Guest</Text>
          </TouchableOpacity>
        </ScrollView>
      </KeyboardAvoidingView>
    </SafeAreaView>
  )
}
```

```typescript
export function ListingDetailScreen({ route, navigation }) {
  const { id } = route.params
  const { user } = useAuthStore()

  const { data: listing, isLoading } = useQuery({
    queryKey: ['listing', id],
    queryFn: () => api.get(`/listings/${id}`).then(r => r.data.data),
  })

  const { data: auction } = useQuery({
    queryKey: ['auction', 'for-listing', id],
    queryFn: () => api.get(`/auctions?listing_id=${id}`).then(r => r.data.data?.[0]),
    enabled: listing?.type === 'auction',
  })

  if (isLoading) return <LoadingSpinner />

  const isAuction = listing?.type === 'auction'
  const price = isAuction ? auction?.current_bid || auction?.start_price : listing?.price

  return (
    <SafeAreaView style={{ flex: 1, backgroundColor: Colors.background }}>

      {/* Header */}
      <View style={{
        flexDirection: 'row', alignItems: 'center',
        backgroundColor: Colors.primary,
        paddingHorizontal: 16, paddingVertical: 12,
```

```typescript
// src/screens/SearchScreen.tsx
export function SearchScreen({ navigation }) {
  const [query, setQuery] = useState('')
  const [debouncedQuery, setDebouncedQuery] = useState('')

  useEffect(() => {
    const t = setTimeout(() => setDebouncedQuery(query), 500)
    return () => clearTimeout(t)
  }, [query])

  const { data, isLoading } = useQuery({
    queryKey: ['search', debouncedQuery],
    queryFn: () => api.get(`/listings?q=${debouncedQuery}&per_page=20`).then(r => r.data.data),
    enabled: debouncedQuery.length > 0,
  })

  return (
    <SafeAreaView style={{ flex: 1, backgroundColor: Colors.background }}>
      {/* Search Header */}
      <View style={{ backgroundColor: Colors.primary, padding: 16 }}>
        <View style={{ flexDirection: 'row', alignItems: 'center', gap: 12 }}>
          <TouchableOpacity onPress={() => navigation.goBack()}>
            <Ionicons name="arrow-back" size={24} color={Colors.white} />
          </TouchableOpacity>
          <View style={{
            flex: 1, backgroundColor: Colors.white,
            borderRadius: 8, paddingHorizontal: 12,
            flexDirection: 'row', alignItems: 'center',
          }}>
            <Ionicons name="search" size={18} color={Colors.gray} />
            <TextInput
              autoFocus
              value={query}
              onChangeText={setQuery}
              placeholder="Search for anything..."
              style={{ flex: 1, padding: 10, fontSize: 15 }}
            />
            {query.length > 0 && (
              <TouchableOpacity onPress={() => setQuery('')}>
                <Ionicons name="close-circle" size={18} color={Colors.gray} />
              </TouchableOpacity>
            )}
          </View>
        </View>
      </View>

      {/* Results — List view like eBay */}
      {isLoading ? (
        <ActivityIndicator style={{ marginTop: 40 }} color={Colors.primary} />
      ) : (
        <FlatList
          data={data || []}
          keyExtractor={(item) => item.id}
          ListEmptyComponent={
            <View style={{ alignItems: 'center', marginTop: 60 }}>
              <Text style={{ fontSize: 40 }}>🔍</Text>
              <Text style={{ color: Colors.gray, marginTop: 8 }}>
                {debouncedQuery ? 'No results found' : 'Start typing to search'}
              </Text>
            </View>
          }
          renderItem={({ item }) => (
            // LIST VIEW for search (not grid)
            <TouchableOpacity
              style={{
                flexDirection: 'row',
                backgroundColor: Colors.white,
                padding: 12, marginBottom: 1,
                gap: 12,
              }}
              onPress={() => navigation.navigate('ListingDetail', { id: item.id })}
            >
              <Image
                source={{ uri: item.images?.[0]?.url || `https://picsum.photos/100?random=${item.id}` }}
                style={{ width: 100, height: 100, borderRadius: 8 }}
                contentFit="cover"
              />
              <View style={{ flex: 1 }}>
                <View style={{ flexDirection: 'row', gap: 6, marginBottom: 4 }}>
                  <View style={{
                    backgroundColor: item.type === 'auction' ? Colors.danger : Colors.primary,
                    borderRadius: 3, paddingHorizontal: 5, paddingVertical: 2,
                  }}>
                    <Text style={{ color: Colors.white, fontSize: 9, fontWeight: 'bold' }}>
                      {item.type === 'auction' ? '🔨 AUCTION' : '⚡ BUY NOW'}
                    </Text>
                  </View>
                </View>
                <Text numberOfLines={2} style={{ fontSize: 14, fontWeight: '500', color: Colors.dark }}>
                  {item.title}
                </Text>
                <Text style={{ fontSize: 12, color: Colors.gray, marginTop: 2 }}>
                  {item.condition}
                </Text>
                <Text style={{ fontSize: 17, fontWeight: 'bold', color: Colors.primary, marginTop: 4 }}>
                  {item.currency} {Number(item.price).toLocaleString()}
                </Text>
                {item.type === 'auction' && <CountdownTimer endsAt={item.ends_at} />}
                <Text style={{ fontSize: 12, color: Colors.gray, marginTop: 2 }}>
                  📍 {item.city}
                </Text>
              </View>
              <TouchableOpacity style={{ padding: 4 }}>
                <Ionicons name="heart-outline" size={22} color={Colors.gray} />
              </TouchableOpacity>
            </TouchableOpacity>
          )}
        />
      )}
    </SafeAreaView>
  )
}
```
