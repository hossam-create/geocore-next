```go
   func SearchListings(ctx context.Context, req SearchRequest) ([]Listing, int64, error) {
       query := db.Model(&Listing{}).
           Where("status = ?", "active")

       // Full-text search
       if req.Query != "" {
           query = query.Where(
               "search_vector @@ plainto_tsquery('english', ?)", 
               req.Query,
           ).Order(gorm.Expr(
               "ts_rank(search_vector, plainto_tsquery('english', ?)) DESC", 
               req.Query,
           ))
       }

       // Price range
       if req.MinPrice != nil {
           query = query.Where("price >= ?", *req.MinPrice)
       }
       if req.MaxPrice != nil {
           query = query.Where("price <= ?", *req.MaxPrice)
       }

       // Category
       if req.CategoryID != nil {
           query = query.Where("category_id = ?", *req.CategoryID)
       }

       // Location (radius search using PostGIS)
       if req.Latitude != nil && req.Longitude != nil {
           radius := 10 // default 10km
           if req.Radius != nil {
               radius = *req.Radius
           }
           query = query.Where(
               "ST_DWithin(ST_MakePoint(longitude, latitude)::geography, ST_MakePoint(?, ?)::geography, ?)",
               *req.Longitude, *req.Latitude, radius*1000, // convert km to meters
           )
       }

       // Sorting
       switch req.SortBy {
       case "price_asc":
           query = query.Order("price ASC")
       case "price_desc":
           query = query.Order("price DESC")
       case "date":
           query = query.Order("created_at DESC")
       case "distance":
           if req.Latitude != nil && req.Longitude != nil {
               query = query.Order(gorm.Expr(
                   "ST_Distance(ST_MakePoint(longitude, latitude)::geography, ST_MakePoint(?, ?)::geography)",
                   *req.Longitude, *req.Latitude,
               ))
           }
       default: // relevance
           if req.Query == "" {
               query = query.Order("created_at DESC")
           }
       }

       // Count total
       var total int64
       query.Count(&total)

       // Pagination
       offset := (req.Page - 1) * req.PerPage
       query = query.Offset(offset).Limit(req.PerPage)

       // Preload associations
       query = query.Preload("Images").Preload("Category")

       var listings []Listing
       err := query.Find(&listings).Error

       return listings, total, err
   }
```

```go
   func (s *NotificationService) Send(userID int64, notif Notification) error {
       // 1. Check user preferences
       prefs, err := s.GetUserPreferences(userID)
       if err != nil {
           return err
       }

       // 2. Create in-app notification
       if prefs.InAppEnabled {
           err = s.createInAppNotification(userID, notif)
           if err != nil {
               log.Error("Failed to create in-app notification", err)
           }
       }

       // 3. Send email (async)
       if s.shouldSendEmail(notif.Type, prefs) {
           s.queueEmailNotification(userID, notif)
       }

       // 4. Send push (async)
       if s.shouldSendPush(notif.Type, prefs) {
           s.queuePushNotification(userID, notif)
       }

       // 5. Send via WebSocket (real-time)
       s.broadcastWebSocket(userID, notif)

       return nil
   }
```
